SetUseVanillaPortrait = {
    effect = {
        if = {
            limit = { NOT = { has_variable = useVanillaPortrait } }
            set_variable = { name = useVanillaPortrait value = 1 }
        }
        else = {
            remove_variable = useVanillaPortrait
        }
    }
}
