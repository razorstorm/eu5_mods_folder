# form_new_culture.txt
INJECT:form_new_culture = {
	allow = {
		in_civil_war = no
		OR = {
			country_rank = country_rank:rank_kingdom
			country_rank = country_rank:rank_empire
		}

		trigger_if = {
			limit = { is_ai = yes }
			culture = { NOT = { has_culture_with_tag = native_american_tag } }
			OR = {
				AND = {
					is_subject_type = colonial_nation
					is_disloyal_subject = yes
				}
			}
		}
	}
}
