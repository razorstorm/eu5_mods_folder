		scope:actor = {
			is_subject = yes

		}
	}
}

INJECT:give_colony_representation = {
	potential = {
		scope:actor ?= {
			any_subject = {
				is_subject_type = colonial_nation
			}
		}
	}
	select_trigger = {
		looking_for_a = country
		target_flag = target_country
		name = "choose_country"
		none_available_msg_key = "no_valid_countries"
		column = {
			data = name
		}
		visible = {
			is_subject_type = colonial_nation
			overlord = scope:actor
			has_policy = colonial_representation_law_no
		}
	}
}

# general_religion.txt
INJECT:convert_religion = {
	allow = {
		trigger_if = {
			limit = {
				exists = scope:actor
			}
			scope:actor = {
				modifier:blocked_from_conversion = no
				NOT = {
					is_subject_type = colonial_nation
				}
			}
		}
	}
}
