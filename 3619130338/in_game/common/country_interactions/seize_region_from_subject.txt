seize_region_from_subject = {
	type = subject
	category = CATEGORY_SUBJECT_ACTIONS
	ai_tick = never
	use_enroute = no
	potential = {
		scope:actor = {
			country_type = location
		}
	}
	allow = {
	}
	select_trigger = {
		looking_for_a = country
		interaction_source_list = {
			scope:actor = {
				every_subject = {
					add_to_list = source
				}
			}
		}
		target_flag = recipient
		name = "choose_country"
		column = {
			data = name
		}
		visible = {
			is_subject_type = colonial_nation
			country_type = location
		}
		enabled = {
			subject_loyalty >= MIN_SUBJECT_LOYALTY_FOR_ACTION
			liberty_desire < MAX_SUBJECT_LIBERTY_DESIRE_FOR_ACTION
		}
	}
	select_trigger = {
		looking_for_a = region
		source = recipient
		target_flag = target
		name = "seize_region_from_subject_select_title"
		none_available_msg_key = "seize_region_from_subject_no_regions"
		column = {
			data = name
		}
		column = {
			data = population
		}
		visible = {
			any_location_in_region = {
				owner ?= scope:recipient
			}
		}
	}
	effect = {
		if = {
			limit = {
				exists = scope:target
			}
			scope:recipient = {
				every_owned_location = {
					limit = {
						region = scope:target
					}
					province = {
						change_province_owner = scope:actor
					}
				}
			}
		}
		else = {
			custom_tooltip = select_a_region_to_take
		}
	}
}
